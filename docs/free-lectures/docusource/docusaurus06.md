## sidebars 추가 

```js
// sidebars.js

/** @type {import('@docusaurus/plugin-content-docs').SidebarsConfig} */
const sidebars = {
  // 기본설정
  tutorialSidebar: [{type: 'autogenerated', dirName: '.'}],
  // 특정 폴더 > 카테고리 
  feGroup: [
    {
      type: "autogenerated",
      dirName: "g-fe-docs",
    },
  ],
  beGroup: [
    {
      type: "autogenerated",
      dirName: "g-be-docs",
    },
  ],
  // But you can create a sidebar manually
};

module.exports = sidebars;

```
